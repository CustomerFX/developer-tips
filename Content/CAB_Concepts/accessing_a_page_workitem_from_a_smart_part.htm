Accessing a Page WorkItem from a Smart Part

Quick forms have a page WorkItem property. For a custom Smart Part, use the Page WorkItem Locator service.

 

 

using Sage.Platform.Application.UI.Web;

public class MyModule : IModule

{

    private IPageWorkItemLocator _pageWorkItemLocator;

    

    [ServiceDependency]

    public IPageWorkItemLocator PageWorkItemLocator

    {

        get { return _pageWorkItemLocator; }

        set { _pageWorkItemLocator = value; }

    }

   

-or-

 

IPageWorkItemLocator locatorService = Sage.Platform.Application.ApplicationContext.Current.Services.Get<IPageWorkItemLocator>();
